def calc_bmi(height,weight):
    # BMIを求める。
    height = height / 100
    bmi = weight / (height * height)
    
    return bmi

def calc_fatness(bmi):
    # 肥満度を求める。
    if bmi < 18:
        judge = 0
    elif bmi >= 18 and bmi < 25:
        judge = 1
    else:
        judge = 2
    
    return judge
